uxcore-uploader

Uxcore Uploader

View on GitHub Download .zip Download .tar.gz

安装

npm install uxcore-uploader

引入类库

<!--required-->
<script src="path/to/jquery.min.js" />
<!--optional-->
<script src="path/to/spark-md5.js" />
<script src="path/to/es5-shim/es5-shim.min.js" />
<script src="path/to/es5-shim/es5-sham.min.js" />

导入资源

import Uploader, {Droparea} from 'uxcore-uploader';

Sample1 - 单文件上传

<Uploader multiple={false}
          autoPending={false}
          name='file'
          url='http://test.yanbingbing.com/upload.php' />

Sample2 - 自动上传

<Uploader multiple={false}
          autoPending={true}
          name='file'
          url='http://test.yanbingbing.com/upload.php' />

Sample3 - 图片上传

<Uploader autoPending={false}
          multiple={false}
          accept="images"
          name='file'
          url='http://test.yanbingbing.com/upload.php' />

Sample4 - 多文件上传

<Uploader autoPending={false}
          multiple={true}
          name='file'
          url='http://test.yanbingbing.com/upload.php' />

Sample5 - 拖拽上传

<Droparea autoPending={false}
          multiple={true} capcity={20}
          name='file'
          url='http://test.yanbingbing.com/upload.php' />

Sample6 - 包裹UI-1

<Uploader autoPending={false}
          name='file'
          url='http://test.yanbingbing.com/upload.php'>
    <i className="uxicon icon-upload" />
    <p>点击或将文件拖拽到此区域上传</p>
</Uploader>

Sample7 - 包裹UI-2

<Droparea autoPending={false}
          className="mydroparea"
          multiple={true}
          capcity={20}
          name='file'
          url='http://test.yanbingbing.com/upload.php'>
    <i className="uxicon icon-upload" />
    <p>点击或将文件拖拽到此区域上传</p>
</Droparea>
